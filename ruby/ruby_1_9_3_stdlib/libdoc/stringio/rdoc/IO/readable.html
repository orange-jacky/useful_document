IO::readable

Similar to read, but raises EOFError at end of string instead of returning nil, as well as IO#sysread does.

 
               static VALUE
strio_sysread(int argc, VALUE *argv, VALUE self)
{
    VALUE val = rb_funcall2(self, rb_intern("read"), argc, argv);
    if (NIL_P(val)) {
        rb_eof_error();
    }
    return val;
}
